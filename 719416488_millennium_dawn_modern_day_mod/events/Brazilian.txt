		}
	}

		name = brazil.3.b

		add_stability = 0.075

			limit = { has_idea = industrial_conglomerates }
			visually_display_opinion_fall_industrial_conglomerates = yes
			set_country_flag = current_industrial_conglomerates
			decrease_internal_faction_opinion = yes
		}
	}
}
