			remove_core_of = SHB
		}
		239 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		928 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		238 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		237 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		581 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		584 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		593 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		937 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		938 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		#Make the winner Somalia
		if = {
			limit = { tag = SHB	}
			set_cosmetic_tag = SOM
			set_country_flag = dynamic_flag
			set_country_flag = dynamic_rebel_flag
			add_ideas = intervention_neo_imperialism
			remove_ideas = Non_State_Actor
		}
		add_state_core = 240
		add_state_core = 239
		add_state_core = 238
		add_state_core = 237
		add_state_core = 581
		add_state_core = 584
		add_state_core = 593
		add_state_core = 937
		add_state_core = 938
		set_capital = 593
		#Mark who won
		if = {
			limit = { tag = SHB }
			set_country_flag = Somalia_Civil_War_Islamist_Victory
			news_event = { id = SomaliaNews.0 hours = 6 }
		}
		if = {
			limit = { tag = SOM }
			set_country_flag = Somalia_Civil_War_Government_Victory
			news_event = { id = SomaliaNews.1 hours = 6 }
			subtract_from_variable = { threat_level = 30 }
			clamp_threat = yes
		}
	}

	option = {
		name = Somalia.0.a
		log = "[GetDateText]: [This.GetName]: Somalia.0.a executed"
	}
}
