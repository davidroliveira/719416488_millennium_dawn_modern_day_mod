		days = 5

		name = usa_economic_events.35.a

			limit = { has_idea = USA_economic_crisis_medium_idea }
			remove_ideas = USA_economic_crisis_medium_idea
		}
		else_if = {
			limit = { has_idea = USA_economic_crisis_severe_idea }

			factor = 1
