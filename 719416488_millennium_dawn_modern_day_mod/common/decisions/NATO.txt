	leave_NATO = {

		icon = GFX_decision_generic_leave_nato

		available = {
			has_idea = NATO_member
		}

		visible = {
			has_idea = NATO_member
		}

		cost = 100

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision leave_NATO"
			remove_ideas = NATO_member
			remove_from_tech_sharing_group = NATO_Tech_Share
			every_other_country = {
				limit = { has_idea = NATO_member }
				add_opinion_modifier = { target = ROOT modifier = left_alliance }
			}
			hidden_effect = {
				news_event = { id = NATO.6 hours = 6 }
			}
		}

		ai_will_do = {
			factor = 0
			modifier = {
				add = 100
				has_government = nationalist
				NOT = {
					tag = HUN
					tag = FYR #North Macedonia
				}
			}
			modifier = {
				add = 1000
				NOT = {
					original_tag = TUR #Turks
					original_tag = HUN #Hungary
					original_tag = GRE #Greece
					original_tag = EST #Estonia
					original_tag = LAT #Latvia
					original_tag = LIT #Lithuania
				}
				OR = {
					AND = {
						NOT = { has_government = democratic }
						NOT = { has_government = neutrality }
					}
					AND = {
						USA = { has_idea = NATO_member }
						NOT = { tag = USA }
						has_opinion = { target = USA value < -100 }
					}
				}
			}
			modifier = {
				factor = 0
				OR = {
					original_tag = TUR #Turks
					original_tag = HUN #Hungary
					original_tag = GRE #Greece
					original_tag = EST #Estonia
					original_tag = LAT #Latvia
					original_tag = LIT #Lithuania
				}
				OR = {
					OR = {
						has_government = neutrality
						has_government = nationalist
						has_government = democratic
						has_government = communism
					}
					OR = {
						if = {
							limit = { USA = { has_idea = NATO_member } }
							has_opinion = { target = USA value > 25 }
						}
						if = {
							limit = { ENG = { has_idea = NATO_member } }
							has_opinion = { target = ENG value > 25 }
						}
						if = {
							limit = { FRA = { has_idea = NATO_member } }
							has_opinion = { target = FRA value > 25 }
						}
						if = {
							limit = { GER = { has_idea = NATO_member } }
							has_opinion = { target = GER value > 25 }
						}
						if = {
							limit = { ITA = { has_idea = NATO_member } }
							has_opinion = { target = ITA value > 25 }
						}
						if = {
							limit = { SPR = { has_idea = NATO_member } }
							has_opinion = { target = SPR value > 25 }
						}
					}

				}
			}
		}
	}

			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision join_NATO"

				}

						has_idea = NATO_member

					limit = { has_idea = Major_Non_NATO_Ally }

				}
			}
			hidden_effect = { news_event = { id = NATO.8 hours = 6 } }
		}

		}

	call_nato_to_arms = {

		icon = GFX_decision_generic_join_nato

		visible = {
			has_defensive_war = yes
			has_idea = NATO_member
			NOT = { has_global_flag = NATO_called_to_arms }
		}

		available = {
			has_idea = NATO_member
			has_defensive_war = yes
			custom_trigger_tooltip = {
				tooltip = TT_ATTACKER_CONTROLS_OUR_STATES
				any_enemy_country = {
					has_offensive_war_with = ROOT
					any_state = {
						is_controlled_by = PREV
						is_owned_by = ROOT
					}
				}
			}
		}

		fire_only_once = yes

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision call_nato_to_arms"
			create_faction = NATO
			set_global_flag = NATO_called_to_arms
			set_country_flag = Leader_of_NATO
			every_other_country = {
				limit = {
					has_idea = NATO_member
					NOT = { has_war_with = ROOT }		#Just in case someone starts a war between NATO countries
					is_subject = no
					is_in_faction = no
				}
				country_event = NATO.3
			}
			hidden_effect = {
				random_country = {
					limit = {
						has_offensive_war_with = ROOT
						any_state = {
							is_controlled_by = PREV
							is_owned_by = ROOT
						}
					}
					set_country_flag = Attacked_NATO
				}
			}
		}

		ai_will_do = {
			factor = 10
		}

	}
